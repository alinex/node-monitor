    ¶
    Rule Definition
  




        		
        

rule:

  fail:
    status: fail
    attempt: 3
    latency: 60
    dependskip: true
    email:
      base: default
    redo: 3h

  warn:
    status: warn
    attempt: 3
    latency: 60
    dependskip: true
    email:
      base: default
    redo: 3h

  ok:
    status: ok
    email:
      base: default




        		
      

      

        


  

    ¶
    When to cleanup entries from storage
  






The values are the number of max. entries of given interval.

        		
        
storage:
  cleanup:
    minute: 360 # 6 hours
    hour: 96 # 4 days
    day: 90 # 3 months
    week: 104 # two years
    month: 60 # 5 years
