exports.init = (cb) ->
  conf = config.get '/monitor'
  debug "Initialize database store..."
  return cb() unless conf.storage?
  database.instance conf.storage.database, (err, db) ->
    return cb err if err
    drop conf, db, (err) ->
      return cb err if err
      create conf, db, cb

drop = (conf, db, cb) ->
  async.eachSeries [
    "DROP SCHEMA public CASCADE"
    "CREATE SCHEMA public"
  ], (sql, cb) ->
    db.exec sql, cb
  , cb
