fail = (err) ->
  if err

    process.exit 1




        		
      

      

        


  

    ¶
    Main routine
  




        		
        
console.log logo
monitor.setup argv._

console.log "Initializing..."
monitor.init (err) ->
  fail err
  conf = config.get 'monitor'
  if argv.info
    console.log 'Not implemented!'
  else if argv.list
    list conf
  else if argv.tree or argv.reverse
    tree conf
  else if argv.daemon
    monitor.start()
    monitor.on 'done', (ctrl) ->
  else
    monitor.on 'result', (ctrl) ->
      console.log chalk.grey "Controller #{chalk.white ctrl.name} => #{ctrl.colorStatus()}"
    console.log "Analyzing systems..."
    monitor.onetime
      verbose: argv.verbose
    , (err, results) ->
      fail err
      console.log "Finished.\n"
