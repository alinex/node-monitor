specific setting.


It's an alinex-validator
compatible schema definition:


        		
        
          
exports.schema =
  title: "Hardware Explorer"
  description: "the configuration to check the hardware of a server"
  type: 'object'
  allowedKeys: true
  keys:
    remote:
      title: "Remote Server"
      description: "the remote server on which to run the command"
      type: 'string'

    explorer: this
    config: config
    result: {}
  explorer.start work
